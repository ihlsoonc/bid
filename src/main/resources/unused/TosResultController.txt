    /*----session value set/get ---------------------------------------*/
    @RequestMapping(value="/si_set_session_value.do")
    public @ResponseBody Map<String,Object>  si_set_session_value(
        @RequestBody Map<String,Object> cmdMap,
        HttpServletRequest request) throws Exception
    {
		HttpSession session = request.getSession();

        for (String key:cmdMap.keySet()) {
		    session.setAttribute(key, cmdMap.get(key));
        }

		HashMap<String, Object> result = new HashMap();
		result.put("result", "Success");  
		return result;
    }
    @RequestMapping(value="/si_get_session_value.do")
    public @ResponseBody Map<String,Object>  si_get_session_value(
        @RequestBody Map<String,Object> cmdMap,
        HttpServletRequest request) throws Exception
    {
		HashMap<String, Object> result = new HashMap();
		HttpSession session = request.getSession();
        for (String key:cmdMap.keySet()) {
		    result.put(key, session.getAttribute(key));
        }
		return result;
    }
